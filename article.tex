\documentclass[a4paper,11pt,oneside,DIV=12,twocolumn]{scrartcl}

    oldstylenums,

    %veryoldstyle,             % use long s (normal with s=)

\usepackage{titlesec}          % customize section headings
\usepackage{soul}              % add space to text

% the use of KOMA-Script and titlesec together is not recommended, 
% but it makes things a lot easier with the soul package 

%https://ctan.space-pro.be/tex-archive/macros/generic/soul/soul-ori.pdf

\newcommand*\periodafter[2]{#1{#2}.}
\titleformat{\section}[block]

	linktoc=page,			   %	- 
	colorlinks=false,	       % 	- colorize links?
	pdfborder={0 0 0},		   % 	-
	breaklinks=true,		   % 	- line break inside links
}

}}
